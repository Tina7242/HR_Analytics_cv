-Cleaning Data in SQL Queries
SELECT
  EmployeeNumber,
  Age,
  CAST(Attrition AS INT64) AS Attrition,
  BusinessTravel,
  DailyRate,
  Department,
  DistanceFromHome,
  Education,
  EducationField,
  EmployeeCount,
  EnvironmentSatisfaction,
  Gender,
  HourlyRate,
  JobInvolvement,
  JobLevel,
  JobRole,
  JobSatisfaction,
  MaritalStatus,
  MonthlyIncome,
  MonthlyRate,
  IFNULL(NumCompaniesWorked, 0) AS NumCompaniesWorked,
  CAST(Over18 AS INT64) AS Over18,
  CAST(OverTime AS INT64) AS OverTime,
  PercentSalaryHike,
  PerformanceRating,
  RelationshipSatisfaction,
  StandardHours,
  StockOptionLevel,
  IFNULL(TotalWorkingYears, 0) AS TotalWorkingYears,
  TrainingTimesLastYear,
  WorkLifeBalance,
  YearsAtCompany,
  YearsInCurrentRole,
  IFNULL(YearsSinceLastPromotion, 0) AS YearsSinceLastPromotion,
  IFNULL(YearsWithCurrManager, 0) AS YearsWithCurrManager
FROM `HR_Analytics_CV.employee_cleaned`;
